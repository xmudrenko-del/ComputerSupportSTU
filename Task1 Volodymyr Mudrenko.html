1.1 Atomic Types

# Numeric values
population <- 8.7
temperature <- 23.5
size_km <- 14500
rating <- 9.8

# Integer values
num_moons <- 3L
num_cities <- 25L
num_species <- 350L

# Logical values
has_water <- TRUE
is_inhabited <- TRUE

# Character values
planet_name <- "Aurelia"
region <- "Andromeda"
category <- "Terrestrial"

cat(planet_name, "is a", category, "planet in", region,
    "with population", population, "billion and temperature", temperature, "°C.\n")
Aurelia is a Terrestrial planet in Andromeda with population 8.7 billion and temperature 23.5 °C.

1.6 Data Frame

  name = c("Aqua", "Lumin", "Orion", "Zephyr", "Terra"),
  population = c(120, 90, 300, 50, 200),
